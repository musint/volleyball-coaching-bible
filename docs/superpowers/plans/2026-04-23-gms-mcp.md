# gms-mcp Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a second MCP server, `gms-mcp`, that ingests Gold Medal Squared (Circle.so) community content into this wiki's `raw/` tree using the Circle internal JSON API, with a 5-tool surface mirroring `aoc-mcp` where sensible.

**Architecture:** Session-cookie authentication via Cookie-Editor JSON import (no Playwright for auth). httpx-based client wraps Circle's `/api/v1/*` JSON endpoints. Post body HTML (Tiptap/ProseMirror) converted to markdown with `markdownify`. Content-type detection via lightweight heuristics. Output files are source-prefixed markdown in shared `raw/articles/`, `raw/transcripts/`, and a new `raw/pdfs/` directory.

**Tech Stack:** Python 3.11+, `mcp>=1.0.0` SDK (stdio), `httpx`, `beautifulsoup4`, `markdownify`, `pydantic>=2`, `playwright` (only for video transcript extraction, copied from aoc-mcp), `pytest`.

**Spec:** `docs/superpowers/specs/2026-04-23-gms-mcp-design.md`

**Pre-resolved open items from spec:**
- HTML → markdown: `markdownify` (matches aoc-mcp)
- Search result cap: 50 (no pagination in v1)
- Mux transcript URL pattern: try `https://stream.mux.com/<playback_id>/text/subtitles.vtt`; on 404 or non-VTT response, emit `transcript_unavailable: true` with provider+id recorded for manual retry. Fixture in Task 11 verifies pattern; update if live capture shows different shape.

```python
def cookies_path() -> Path:
    """Resolve the Cookie-Editor JSON path from env; raise if missing."""
    raw = os.environ.get("GMS_COOKIES_PATH")
    if not raw:
        raise RuntimeError(
            "GMS_COOKIES_PATH env var not set. "
            "Export cookies with the Cookie-Editor browser extension "
            "(community.goldmedalsquared.com), then set GMS_COOKIES_PATH "
            "to the absolute path of the exported JSON file."
        )
    return Path(raw).expanduser().resolve()


def repo_root() -> Path:
    """Walk up from the module until we find a dir containing `raw/`."""
    here = Path(__file__).resolve()
    for parent in (here, *here.parents):
        if (parent / "raw").is_dir() and (parent / "wiki").is_dir():
            return parent
    raise RuntimeError("Could not locate repo root (no sibling `raw/` and `wiki/`)")


def raw_dir(subdir: str) -> Path:
    """Return `<repo-root>/raw/<subdir>`, creating it if needed."""
    p = repo_root() / "raw" / subdir
    p.mkdir(parents=True, exist_ok=True)
    return p
```

```python
"""Structured error hierarchy for gms-mcp."""


class GmsError(Exception):
    code: str = "gms_error"


class AuthMissing(GmsError):
    code = "auth_missing"


class AuthExpired(GmsError):
    code = "auth_expired"

    def __init__(self, message: str | None = None):
        super().__init__(
            message
            or "GMS session expired. Re-export cookies with the Cookie-Editor "
            "browser extension and update GMS_COOKIES_PATH."
        )


class NotFound(GmsError):
    code = "not_found"

    def __init__(self, url: str):
        self.url = url
        super().__init__(f"{url} returned 404")


class RateLimited(GmsError):
    code = "rate_limited"

    def __init__(self, url: str):
        self.url = url
        super().__init__(f"{url} rate-limited (429) after retry")


class UpstreamError(GmsError):
    code = "upstream_error"

    def __init__(self, url: str, status: int):
        self.url = url
        self.status = status
        super().__init__(f"{url} upstream error ({status}) after retry")


class ParserFailed(GmsError):
    code = "parser_failed"

    def __init__(self, context: str, snippet: str):
        self.context = context
        self.snippet = snippet[:500]
        super().__init__(f"Parser failed in {context}. Snippet: {self.snippet!r}")


class UnknownSpace(GmsError):
    code = "unknown_space"

    def __init__(self, slug: str, known: list[str]):
        self.slug = slug
        self.known = known
        super().__init__(
            f"Unknown space slug '{slug}'. Known: {', '.join(sorted(known))}"
        )


class AttachmentTooLarge(GmsError):
    code = "attachment_too_large"

    def __init__(self, filename: str, size: int, url: str):
        self.filename = filename
        self.size = size
        self.url = url
        super().__init__(
            f"Attachment {filename} is {size} bytes (exceeds guard). "
            f"Signed URL for manual retrieval: {url}"
        )


class UnsupportedVideoProvider(GmsError):
    code = "unsupported_video_provider"

    def __init__(self, provider: str, url: str):
        self.provider = provider
        self.url = url
        super().__init__(f"Unsupported video provider '{provider}' at {url}")
```

```python
"""Load cookies exported from the Cookie-Editor browser extension."""
from __future__ import annotations

import json
from pathlib import Path

import httpx

from gms_mcp.config import COMMUNITY_HOST
from gms_mcp.errors import AuthMissing


def _matches_gms_domain(cookie_domain: str) -> bool:
    """Accept cookies for community.goldmedalsquared.com or .goldmedalsquared.com."""
    d = cookie_domain.lstrip(".")
    return d == COMMUNITY_HOST or COMMUNITY_HOST.endswith("." + d)


def load_cookies(path: Path) -> httpx.Cookies:
    """Parse a Cookie-Editor JSON export and return httpx.Cookies filtered to GMS.

    Raises AuthMissing if the file is missing, malformed, or contains no cookies
    for community.goldmedalsquared.com.
    """
    if not path.exists():
        raise AuthMissing(
            f"Cookie file not found at {path}. "
            "Export cookies via Cookie-Editor extension for "
            f"https://{COMMUNITY_HOST} and point GMS_COOKIES_PATH at the file."
        )
    try:
        raw = json.loads(path.read_text(encoding="utf-8"))
    except (json.JSONDecodeError, UnicodeDecodeError) as e:
        raise AuthMissing(f"Cookie file at {path} is malformed: {e}") from e

    if not isinstance(raw, list):
        raise AuthMissing(
            f"Cookie file at {path} is not a JSON array. "
            "Use the Cookie-Editor 'Export' → 'JSON' option."
        )

    cookies = httpx.Cookies()
    matched = 0
    for entry in raw:
        domain = entry.get("domain", "")
        if not _matches_gms_domain(domain):
            continue
        cookies.set(
            name=entry["name"],
            value=entry["value"],
            domain=domain,
            path=entry.get("path", "/"),
        )
        matched += 1

    if matched == 0:
        raise AuthMissing(
            f"No cookies for {COMMUNITY_HOST} found in {path}. "
            "Make sure you exported cookies while logged in to the GMS community."
        )
    return cookies
```

```python
"""Authenticated httpx client for Circle /api/v1/*. Handles rate-limiting,
retries, and session-expiry detection."""
from __future__ import annotations

import time
from pathlib import Path
from typing import Iterator
from urllib.parse import urljoin

import httpx

from gms_mcp import config
from gms_mcp.auth import load_cookies
from gms_mcp.errors import AuthExpired, NotFound, RateLimited, UpstreamError


class GmsClient:
    """Session-authenticated client for community.goldmedalsquared.com.

    Pass _transport for tests; otherwise a real httpx.HTTPTransport is used.
    """

    def __init__(
        self,
        cookies_path: Path | None = None,
        *,
        _transport: httpx.BaseTransport | None = None,
    ):
        path = cookies_path or config.cookies_path()
        cookies = load_cookies(path)
        self._http = httpx.Client(
            base_url=config.COMMUNITY_BASE_URL,
            cookies=cookies,
            headers={
                "User-Agent": config.USER_AGENT,
                "Accept": "application/json",
            },
            timeout=config.DEFAULT_TIMEOUT_S,
            follow_redirects=True,
            transport=_transport,
        )
        self._last_request = 0.0

    def _rate_limit(self) -> None:
        deadline = self._last_request + config.REQUEST_MIN_INTERVAL_S
        while True:
            remaining = deadline - time.monotonic()
            if remaining <= 0:
                return
            time.sleep(remaining)

    def _absolute(self, path_or_url: str) -> str:
        if path_or_url.startswith("http://") or path_or_url.startswith("https://"):
            return path_or_url
        return urljoin(config.COMMUNITY_BASE_URL, path_or_url)

    def get_json(self, path_or_url: str, *, params: dict | None = None) -> dict:
        url = self._absolute(path_or_url)
        self._rate_limit()
        self._last_request = time.monotonic()
        response = self._http.get(url, params=params)

        if response.status_code == 401 or response.status_code == 403:
            raise AuthExpired()
        if response.status_code == 404:
            raise NotFound(url)
        if response.status_code == 429:
            time.sleep(config.RETRY_429_BACKOFF_S)
            response = self._http.get(url, params=params)
            if response.status_code == 429:
                raise RateLimited(url)
        if 500 <= response.status_code < 600:
            time.sleep(config.RETRY_5XX_BACKOFF_S)
            response = self._http.get(url, params=params)
            if 500 <= response.status_code < 600:
                raise UpstreamError(url, response.status_code)
        response.raise_for_status()
        return response.json()

    def paginate(
        self,
        path_or_url: str,
        *,
        key: str,
        params: dict | None = None,
        max_pages: int = 100,
    ) -> Iterator[dict]:
        """Yield items from `key` across pages until an empty page is returned.

        Circle uses `?page=N`. We stop when the list is empty or we hit max_pages.
        """
        base_params = dict(params or {})
        for page in range(1, max_pages + 1):
            page_params = {**base_params, "page": page}
            data = self.get_json(path_or_url, params=page_params)
            items = data.get(key, [])
            if not items:
                return
            yield from items

    def download_bytes(self, url: str, *, max_bytes: int) -> bytes:
        """Download up to max_bytes; raise if the resource is larger."""
        self._rate_limit()
        self._last_request = time.monotonic()
        with self._http.stream("GET", url) as resp:
            if resp.status_code == 404:
                raise NotFound(url)
            resp.raise_for_status()
            chunks: list[bytes] = []
            total = 0
            for chunk in resp.iter_bytes():
                total += len(chunk)
                if total > max_bytes:
                    resp.close()
                    raise ValueError(f"Resource at {url} exceeds {max_bytes} bytes")
                chunks.append(chunk)
            return b"".join(chunks)

    def close(self) -> None:
        self._http.close()

    def __enter__(self) -> "GmsClient":
        return self

    def __exit__(self, exc_type, exc, tb) -> None:
        self.close()
```

```python
"""One-shot live capture. Run manually with GMS_COOKIES_PATH set:

    python -m gms_mcp.discover --all

Outputs JSON fixtures into tests/fixtures/ for replay tests. Use --space SLUG
and --post URL to capture specific examples for each content type.

from __future__ import annotations

import argparse
import json
import sys
from pathlib import Path
from urllib.parse import urlparse

from gms_mcp.client import GmsClient
from gms_mcp import config

FIXTURE_DIR = Path(__file__).parent.parent / "tests" / "fixtures"


def save(name: str, data) -> Path:
    FIXTURE_DIR.mkdir(parents=True, exist_ok=True)
    path = FIXTURE_DIR / f"{name}.json"
    path.write_text(json.dumps(data, indent=2, default=str), encoding="utf-8")
    print(f"[discover] wrote {path}")
    return path


def capture_me(client: GmsClient) -> None:
    data = client.get_json("/api/v1/community_members/me.json")
    save("me", data)


def capture_spaces(client: GmsClient) -> None:
    data = client.get_json("/api/v1/spaces.json")
    save("spaces", data)


def capture_space_recent(client: GmsClient, space_slug: str) -> None:
    spaces = client.get_json("/api/v1/spaces.json")
    space_list = spaces if isinstance(spaces, list) else spaces.get("spaces", [])
    target = next((s for s in space_list if s.get("slug") == space_slug), None)
    if target is None:
        print(f"[discover] space {space_slug} not found", file=sys.stderr)
        return
    data = client.get_json(
        "/api/v1/posts.json",
        params={"space_id": target["id"], "page": 1},
    )
    save(f"posts_{space_slug}", data)


def _post_slug_from_url(url: str) -> str:
    # .../c/<space>/<post-slug>
    parts = urlparse(url).path.strip("/").split("/")
    return parts[-1] if parts else "post"


def capture_post(client: GmsClient, post_url: str) -> None:
    # Post JSON endpoint accepts the slug; Circle's internal API uses id,
    # so we resolve slug -> id via the space listing first. We accept the
    # full community URL here for convenience.
    parts = urlparse(post_url).path.strip("/").split("/")
    # Expected: c/<space-slug>/<post-slug>
    if len(parts) < 3 or parts[0] != "c":
        print(f"[discover] cannot parse URL {post_url}", file=sys.stderr)
        return
    space_slug, post_slug = parts[1], parts[2]
    # Fetch posts in that space until we find the slug; simplistic but fine for
    # fixture capture. In production, Circle exposes a by-slug endpoint too —
    # discover will reveal the actual shape.
    spaces_data = client.get_json("/api/v1/spaces.json")
    space_list = spaces_data if isinstance(spaces_data, list) else spaces_data.get("spaces", [])
    target_space = next((s for s in space_list if s.get("slug") == space_slug), None)
    if target_space is None:
        print(f"[discover] space {space_slug} not visible to session", file=sys.stderr)
        return
    for page in range(1, 20):
        posts = client.get_json(
            "/api/v1/posts.json",
            params={"space_id": target_space["id"], "page": page},
        )
        post_list = posts if isinstance(posts, list) else posts.get("posts") or posts.get("records") or []
        match = next((p for p in post_list if p.get("slug") == post_slug), None)
        if match:
            post_id = match["id"]
            detail = client.get_json(f"/api/v1/posts/{post_id}.json")
            save(f"post_{post_slug}", detail)
            try:
                comments = client.get_json(f"/api/v1/posts/{post_id}/comments.json")
                save(f"comments_{post_slug}", comments)
            except Exception as e:
                print(f"[discover] comments fetch failed: {e}", file=sys.stderr)
            return
        if not post_list:
            break
    print(f"[discover] post {post_slug} not found in {space_slug}", file=sys.stderr)


def capture_search(client: GmsClient, query: str) -> None:
    data = client.get_json("/api/v1/search", params={"q": query})
    save(f"search_{query.replace(' ', '_')}", data)


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--all", action="store_true", help="Capture baseline: me, spaces")
    parser.add_argument("--space", help="Capture posts list for a space slug")
    parser.add_argument("--post", help="Capture a single post by full URL")
    parser.add_argument("--search", help="Capture search results for a query")
    args = parser.parse_args()

    with GmsClient() as client:
        if args.all:
            capture_me(client)
            capture_spaces(client)
        if args.space:
            capture_space_recent(client, args.space)
        if args.post:
            capture_post(client, args.post)
        if args.search:
            capture_search(client, args.search)
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
"""List and resolve Circle spaces accessible to the current session."""
from __future__ import annotations

from typing import Iterable

from gms_mcp.client import GmsClient
from gms_mcp.errors import UnknownSpace
from gms_mcp.models import Space, SpaceList


def spaces_from_payload(payload) -> list[Space]:
    """Accept either a bare list or a wrapper dict with a 'spaces'/'records' key."""
    if isinstance(payload, list):
        raw = payload
    elif isinstance(payload, dict):
        raw = payload.get("spaces") or payload.get("records") or []
    else:
        raw = []
    spaces: list[Space] = []
    for item in raw:
        spaces.append(
            Space(
                id=item["id"],
                slug=item.get("slug") or item["id"],
                name=item.get("name") or item.get("display_name") or str(item["id"]),
                post_count=(
                    item.get("post_count")
                    or item.get("posts_count")
                    or item.get("topics_count")
                ),
                space_group=(
                    item.get("space_group_name")
                    or (item.get("space_group") or {}).get("name")
                ),
            )
        )
    return spaces


def list_spaces(client: GmsClient) -> SpaceList:
    payload = client.get_json("/api/v1/spaces.json")
    return SpaceList(spaces=spaces_from_payload(payload))


def resolve_space_id(spaces: Iterable[Space], slug: str) -> int:
    matches = [s.id for s in spaces if s.slug == slug]
    if not matches:
        known = [s.slug for s in spaces]
        raise UnknownSpace(slug, known)
    return matches[0]
```

```python
"""Post JSON → markdown file with YAML front-matter.

This module does NOT write to disk; it returns the full file contents as a string.
The server layer handles file placement and collision detection.
"""
from __future__ import annotations

from datetime import datetime, timezone
from typing import Any

import yaml
from dateutil import parser as date_parser
from markdownify import markdownify as md

from gms_mcp import config
from gms_mcp.models import Attachment, ContentType, VideoEmbed


def _iso(dt_str: str | None) -> str:
    if not dt_str:
        return datetime.now(timezone.utc).isoformat()
    dt = date_parser.isoparse(dt_str)
    if dt.tzinfo is None:
        dt = dt.replace(tzinfo=timezone.utc)
    return dt.isoformat()


def _post_url(post: dict) -> str:
    space_slug = post.get("space_slug") or post.get("space", {}).get("slug", "")
    post_slug = post.get("slug", "")
    return f"{config.COMMUNITY_BASE_URL}/c/{space_slug}/{post_slug}"


def _post_body_html(post: dict) -> str:
    body = post.get("body")
    if isinstance(body, dict):
        return body.get("body") or body.get("html") or body.get("body_html") or ""
    if isinstance(body, str):
        return body
    return post.get("body_html") or post.get("tiptap_body") or ""


def body_html_to_markdown(html: str) -> str:
    """Convert Circle's Tiptap/ProseMirror HTML to markdown via markdownify."""
    if not html:
        return ""
    return md(html, heading_style="ATX", bullets="*").strip()


def build_front_matter(
    post: dict,
    *,
    content_type: ContentType,
    attachments: list[Attachment],
    video: VideoEmbed | None,
    comments_included: bool = False,
) -> dict[str, Any]:
    user = post.get("user") or {}
    author = user.get("name") or user.get("display_name") or "Unknown"
    space_slug = post.get("space_slug") or post.get("space", {}).get("slug", "")

    fm: dict[str, Any] = {
        "source": "gms",
        "space": space_slug,
        "space_id": post.get("space_id"),
        "post_id": post["id"],
        "slug": post["slug"],
        "url": _post_url(post),
        "author": author,
        "published_at": _iso(post.get("created_at") or post.get("published_at")),
        "content_type": content_type,
        "has_document": any(
            a.filename.lower().endswith((".pdf", ".docx", ".xlsx", ".pptx"))
            for a in attachments
        ),
        "tags": list(post.get("tag_list") or post.get("tags") or []),
        "attachments": [a.model_dump() for a in attachments],
        "comments_included": comments_included,
        "ingested_at": datetime.now(timezone.utc).date().isoformat(),
    }
    if video is not None:
        fm["video"] = video.model_dump()
    return fm


def extract_post(
    post: dict,
    *,
    content_type: ContentType,
    attachments: list[Attachment],
    video: VideoEmbed | None,
    comments_md: str | None,
) -> str:
    fm = build_front_matter(
        post,
        content_type=content_type,
        attachments=attachments,
        video=video,
        comments_included=bool(comments_md),
    )
    fm_yaml = yaml.safe_dump(fm, sort_keys=False, allow_unicode=True)
    body_html = _post_body_html(post)
    body_md = body_html_to_markdown(body_html)

    parts = [f"---\n{fm_yaml}---\n", f"# {post.get('name') or post.get('title') or ''}\n", body_md]
    if comments_md:
        parts.append("\n## Comments\n")
        parts.append(comments_md)
    return "\n".join(p for p in parts if p).rstrip() + "\n"
```

```python
"""Flatten Circle comment trees into threaded markdown (depth-3 cap)."""
from __future__ import annotations

from gms_mcp.config import COMMENT_MAX_DEPTH
from gms_mcp.extract import body_html_to_markdown


def _comment_body_html(comment: dict) -> str:
    body = comment.get("body")
    if isinstance(body, dict):
        return body.get("body") or body.get("html") or ""
    if isinstance(body, str):
        return body
    return comment.get("body_html") or ""


def _render_one(comment: dict, depth: int) -> list[str]:
    effective = min(depth, COMMENT_MAX_DEPTH)
    prefix = ">" * effective
    user = (comment.get("user") or {}).get("name") or "Unknown"
    when = comment.get("created_at") or ""
    body_md = body_html_to_markdown(_comment_body_html(comment))
    # Collapse body into blockquote-friendly lines
    body_lines = [f"{prefix} {line}".rstrip() for line in (body_md.splitlines() or [""])]
    header = f"{prefix} **{user}** · {when}"
    lines = [header, *body_lines, ""]
    for reply in comment.get("replies") or []:
        lines.extend(_render_one(reply, depth + 1))
    return lines


def _normalize_tree(payload) -> list[dict]:
    """Accept either a list, a 'records'/'comments' wrapped dict, or a nested tree.
    If flat, reattach replies by parent_id if that field exists."""
    if isinstance(payload, list):
        raw = payload
    elif isinstance(payload, dict):
        raw = payload.get("comments") or payload.get("records") or []
    else:
        raw = []

    if not raw:
        return []

    first = raw[0]
    if "replies" in first:
        return raw

    # Build parent→children map from parent_id
    by_id = {c["id"]: {**c, "replies": []} for c in raw}
    roots: list[dict] = []
    for c in raw:
        pid = c.get("parent_id")
        if pid and pid in by_id:
            by_id[pid]["replies"].append(by_id[c["id"]])
        else:
            roots.append(by_id[c["id"]])
    return roots


def render_comments(payload) -> str:
    roots = _normalize_tree(payload)
    if not roots:
        return ""
    out: list[str] = []
    for root in roots:
        out.extend(_render_one(root, depth=1))
    return "\n".join(out).rstrip() + "\n"
```

```python
"""Download Circle post attachments to raw/pdfs/ with a size guard."""
from __future__ import annotations

from pathlib import Path
from typing import Any, Sequence

from gms_mcp import config
from gms_mcp.errors import AttachmentTooLarge
from gms_mcp.models import Attachment


def parse_attachment_entries(raw: Any) -> list[dict]:
    """Normalize Circle's attachment entries to {filename, url, byte_size}.

    Circle's exact shape has variants; accept either 'name'/'url' or
    'filename'/'download_url'/'signed_url'. Unknown entries are dropped.
    """
    if not raw:
        return []
    out: list[dict] = []
    for item in raw:
        filename = item.get("name") or item.get("filename") or item.get("file_name")
        url = (
            item.get("url")
            or item.get("download_url")
            or item.get("signed_url")
            or item.get("file_url")
        )
        if not (filename and url):
            continue
        out.append({
            "filename": filename,
            "url": url,
            "byte_size": item.get("byte_size") or item.get("size_bytes") or item.get("size"),
        })
    return out


def _safe_filename(post_slug: str, filename: str) -> str:
    # keep extension, sanitize basename
    stem = Path(filename).stem
    ext = Path(filename).suffix
    safe_stem = "".join(c if c.isalnum() or c in "-_" else "-" for c in stem).strip("-")
    return f"gms-{post_slug}-{safe_stem}{ext}" if safe_stem else f"gms-{post_slug}{ext}"


def download_attachments(
    client,
    entries: Sequence[dict],
    *,
    post_slug: str,
    raw_pdfs_dir: Path,
) -> list[Attachment]:
    raw_pdfs_dir.mkdir(parents=True, exist_ok=True)
    saved: list[Attachment] = []
    for entry in entries:
        declared = entry.get("byte_size")
        if isinstance(declared, int) and declared > config.ATTACHMENT_MAX_BYTES:
            raise AttachmentTooLarge(entry["filename"], declared, entry["url"])
        try:
            data = client.download_bytes(entry["url"], max_bytes=config.ATTACHMENT_MAX_BYTES)
        except ValueError as e:
            raise AttachmentTooLarge(
                entry["filename"], declared or config.ATTACHMENT_MAX_BYTES + 1, entry["url"]
            ) from e
        filename = _safe_filename(post_slug, entry["filename"])
        path = raw_pdfs_dir / filename
        path.write_bytes(data)
        saved.append(Attachment(
            filename=filename,
            raw_path=str(path),
            source_url=entry["url"],
            size_bytes=len(data),
        ))
    return saved
```

```python
"""Video embed detection + transcript extraction for Mux / Vimeo / Wistia."""
from __future__ import annotations

import re
from typing import Any

import httpx

from gms_mcp import config
from gms_mcp.errors import UnsupportedVideoProvider
from gms_mcp.models import TranscriptResult, VideoProvider


_MUX_RE = re.compile(
    r'<mux-player[^>]*playback-id=["\']([A-Za-z0-9]+)["\']', re.IGNORECASE
)
_VIMEO_RE = re.compile(
    r'player\.vimeo\.com/video/(\d+)', re.IGNORECASE
)
_WISTIA_RE = re.compile(
    r'wistia_async_([A-Za-z0-9]+)', re.IGNORECASE
)
_YOUTUBE_RE = re.compile(
    r'(?:youtube\.com/embed/|youtu\.be/|youtube\.com/watch\?v=)([A-Za-z0-9_-]{11})',
    re.IGNORECASE,
)


def detect_embed(html: str) -> tuple[VideoProvider, str]:
    if not html:
        return "unknown", ""
    for provider, rx in (
        ("mux", _MUX_RE),
        ("vimeo", _VIMEO_RE),
        ("wistia", _WISTIA_RE),
        ("youtube", _YOUTUBE_RE),
    ):
        m = rx.search(html)
        if m:
            return provider, m.group(1)
    return "unknown", ""


def mux_transcript_url(playback_id: str) -> str:
    return f"https://stream.mux.com/{playback_id}/text/subtitles.vtt"


def _vtt_to_plain_text(vtt: str) -> str:
    """Strip VTT headers and cue timings, keep caption text only."""
    lines = []
    for line in vtt.splitlines():
        if not line.strip():
            continue
        if line.strip() == "WEBVTT":
            continue
        if "-->" in line:
            continue
        if re.match(r"^\d+$", line.strip()):
            continue
        if line.startswith("NOTE") or line.startswith("STYLE"):
            continue
        lines.append(line.strip())
    return "\n".join(lines).strip()


def fetch_mux_transcript(
    playback_id: str, *, http: httpx.Client | None = None
) -> str | None:
    url = mux_transcript_url(playback_id)
    client = http or httpx.Client(timeout=config.DEFAULT_TIMEOUT_S)
    try:
        resp = client.get(url)
    finally:
        if http is None:
            client.close()
    if resp.status_code != 200 or "WEBVTT" not in resp.text[:100]:
        return None
    return _vtt_to_plain_text(resp.text)


def fetch_transcript(
    provider: VideoProvider, video_id: str, *, http: httpx.Client | None = None
) -> TranscriptResult:
    if provider == "mux":
        text = fetch_mux_transcript(video_id, http=http)
        return TranscriptResult(
            provider="mux",
            id=video_id,
            transcript_unavailable=text is None,
        ) if text is None else TranscriptResult(
            provider="mux",
            id=video_id,
        )
    if provider in ("vimeo", "wistia", "youtube"):
        # Vimeo/Wistia/YouTube transcript extraction requires Playwright + provider-specific
        # capture logic — reuse aoc-mcp's path on a separate branch if/when needed.
        # For v1 we record the embed id and defer transcript capture.
        return TranscriptResult(
            provider=provider, id=video_id, transcript_unavailable=True
        )
    raise UnsupportedVideoProvider(provider, video_id)
```

```python
"""Wrap Circle's /api/v1/search endpoint into a PostList."""
from __future__ import annotations

from gms_mcp import config
from gms_mcp.client import GmsClient
from gms_mcp.models import PostList, PostSummary


def summary_from_record(record: dict) -> PostSummary:
    user = record.get("user") or {}
    space_slug = record.get("space_slug") or record.get("space", {}).get("slug", "")
    post_slug = record.get("slug", "")
    url = f"{config.COMMUNITY_BASE_URL}/c/{space_slug}/{post_slug}"
    return PostSummary(
        id=record["id"],
        slug=post_slug,
        title=record.get("name") or record.get("title") or "",
        author=user.get("name") or "Unknown",
        published_at=record.get("created_at") or record.get("published_at"),
        space=space_slug,
        url=url,
        comment_count=record.get("comments_count") or 0,
    )


def run_search(
    client: GmsClient, query: str, *, limit: int = 20, space: str | None = None
) -> PostList:
    params = {"q": query}
    if space:
        params["space_slug"] = space
    payload = client.get_json("/api/v1/search", params=params)
    records = payload.get("records") or payload.get("results") or payload.get("posts") or []
    posts = [summary_from_record(r) for r in records]
    if space:
        posts = [p for p in posts if p.space == space]
    capped = min(limit, config.SEARCH_RESULT_CAP)
    return PostList(posts=posts[:capped])
```

```python
def _error_payload(err: GmsError) -> dict[str, Any]:
    payload: dict[str, Any] = {"code": err.code, "message": str(err)}
    for attr in ("url", "slug", "known", "provider", "status", "filename", "size", "context", "snippet"):
        if hasattr(err, attr):
            payload[attr] = getattr(err, attr)
    return payload


def _list_recent_impl(client: GmsClient, *, space: str | None, limit: int, since: str | None) -> dict:
    all_spaces = spaces.list_spaces(client).spaces
    target_spaces = [s for s in all_spaces if (space is None or s.slug == space)]
    if space and not target_spaces:
        from gms_mcp.errors import UnknownSpace
        raise UnknownSpace(space, [s.slug for s in all_spaces])
    posts: list[dict] = []
    for sp in target_spaces:
        payload = client.get_json("/api/v1/posts.json", params={"space_id": sp.id, "page": 1})
        records = payload.get("posts") or payload.get("records") or []
        for r in records:
            r.setdefault("space_slug", sp.slug)
            posts.append(r)
    # Filter `since` if provided
    if since:
        from dateutil import parser as dp
        threshold = dp.isoparse(since)
        posts = [p for p in posts if dp.isoparse(p.get("created_at", "")) >= threshold]
    posts = sorted(posts, key=lambda p: p.get("created_at", ""), reverse=True)[:limit]
    from gms_mcp.search import summary_from_record
    return {"posts": [summary_from_record(p).model_dump() for p in posts]}


def _fetch_post_impl(client: GmsClient, url_or_slug: str) -> dict:
    # Accept full URL or "<space>/<slug>"
    if url_or_slug.startswith("http"):
        parts = urlparse(url_or_slug).path.strip("/").split("/")
        if len(parts) < 3 or parts[0] != "c":
            raise ParserFailed("url_or_slug", url_or_slug)
        space_slug, post_slug = parts[1], parts[2]
    elif "/" in url_or_slug:
        space_slug, post_slug = url_or_slug.split("/", 1)
    else:
        raise ParserFailed("url_or_slug", url_or_slug)

    # Resolve slug → id via space's posts list
    all_spaces = spaces.list_spaces(client).spaces
    target_space = next((s for s in all_spaces if s.slug == space_slug), None)
    if target_space is None:
        from gms_mcp.errors import UnknownSpace
        raise UnknownSpace(space_slug, [s.slug for s in all_spaces])
    post_id = None
    for page in range(1, 20):
        payload = client.get_json(
            "/api/v1/posts.json", params={"space_id": target_space.id, "page": page}
        )
        records = payload.get("posts") or payload.get("records") or []
        match = next((r for r in records if r.get("slug") == post_slug), None)
        if match:
            post_id = match["id"]
            break
        if not records:
            break
    if post_id is None:
        from gms_mcp.errors import NotFound
        raise NotFound(f"{space_slug}/{post_slug}")

    post = client.get_json(f"/api/v1/posts/{post_id}.json")
    post.setdefault("space_slug", space_slug)
    post.setdefault("space_id", target_space.id)

    # Attachments
    raw_pdfs_dir = config.raw_dir("pdfs")
    attachment_entries = attachments.parse_attachment_entries(post.get("attachments") or [])
    saved_attachments = attachments.download_attachments(
        client, attachment_entries, post_slug=post_slug, raw_pdfs_dir=raw_pdfs_dir
    )

    # Video
    body_html = extract._post_body_html(post)
    provider, vid = video.detect_embed(body_html)
    video_embed: VideoEmbed | None = None
    if provider != "unknown":
        transcript_path: str | None = None
        unavailable = True
        if provider == "mux":
            text = video.fetch_mux_transcript(vid)
            if text:
                transcripts_dir = config.raw_dir("transcripts")
                transcript_path = str(transcripts_dir / f"gms-{post_slug}.md")
                Path(transcript_path).write_text(
                    f"# Transcript — {post.get('name')}\n\n{text}\n",
                    encoding="utf-8",
                )
                unavailable = False
        video_embed = VideoEmbed(
            provider=provider,
            id=vid,
            transcript_path=transcript_path,
            transcript_unavailable=unavailable,
        )

    # Comments
    comments_md: str | None = None
    if post.get("comments_count"):
        try:
            payload = client.get_json(f"/api/v1/posts/{post_id}/comments.json")
            comments_md = comments.render_comments(payload) or None
        except GmsError:
            comments_md = None

    # Content type
    content_type = extract.detect_content_type(post)

    # Build markdown
    file_md = extract.extract_post(
        post,
        content_type=content_type,
        attachments=saved_attachments,
        video=video_embed,
        comments_md=comments_md,
    )

    # Write primary file; on slug collision (different post_id), disambiguate
    articles_dir = config.raw_dir("articles")
    primary = articles_dir / f"gms-{post_slug}.md"
    if primary.exists():
        existing = primary.read_text(encoding="utf-8")
        if f"post_id: {post_id}\n" not in existing:
            # different post_id — disambiguate both files
            disambig = articles_dir / f"gms-{space_slug}-{post_slug}.md"
            primary.rename(articles_dir / f"gms-{_space_from_front_matter(existing)}-{post_slug}.md")
            primary = disambig
    primary.write_text(file_md, encoding="utf-8")

    return FetchedPost(
        post_id=post_id,
        slug=post_slug,
        space=space_slug,
        url=f"{config.COMMUNITY_BASE_URL}/c/{space_slug}/{post_slug}",
        raw_path=str(primary),
        attachment_paths=[a.raw_path for a in saved_attachments],
        video=video_embed,
        content_type=content_type,
    ).model_dump()


def _space_from_front_matter(md_text: str) -> str:
    for line in md_text.splitlines():
        if line.startswith("space: "):
            return line.split(":", 1)[1].strip()
    return "unknown"


def _video_transcript_impl(client: GmsClient, url: str) -> dict:
    # If it's a Circle post URL, fetch the post and detect the embed.
    if "community.goldmedalsquared.com" in url:
        parts = urlparse(url).path.strip("/").split("/")
        if len(parts) >= 3 and parts[0] == "c":
            space_slug, post_slug = parts[1], parts[2]
            all_spaces = spaces.list_spaces(client).spaces
            target_space = next((s for s in all_spaces if s.slug == space_slug), None)
            if target_space is None:
                from gms_mcp.errors import UnknownSpace
                raise UnknownSpace(space_slug, [s.slug for s in all_spaces])
            # find post id
            post_id = None
            for page in range(1, 20):
                payload = client.get_json(
                    "/api/v1/posts.json", params={"space_id": target_space.id, "page": page}
                )
                records = payload.get("posts") or payload.get("records") or []
                match = next((r for r in records if r.get("slug") == post_slug), None)
                if match:
                    post_id = match["id"]
                    break
                if not records:
                    break
            if post_id is None:
                from gms_mcp.errors import NotFound
                raise NotFound(f"{space_slug}/{post_slug}")
            post = client.get_json(f"/api/v1/posts/{post_id}.json")
            provider, vid = video.detect_embed(extract._post_body_html(post))
            result = video.fetch_transcript(provider, vid)
            if not result.transcript_unavailable and provider == "mux":
                text = video.fetch_mux_transcript(vid)
                transcripts_dir = config.raw_dir("transcripts")
                path = transcripts_dir / f"gms-{post_slug}.md"
                path.write_text(f"# Transcript — {post.get('name')}\n\n{text}\n", encoding="utf-8")
                result = result.model_copy(update={"transcript_path": str(path)})
            return result.model_dump()
    # Bare video URL: detect provider from URL shape
    provider, vid = video.detect_embed(f'<iframe src="{url}"></iframe>')
    return video.fetch_transcript(provider, vid).model_dump()


@server.call_tool()
async def call_tool(name: str, arguments: dict[str, Any]) -> list[TextContent]:
    try:
        with GmsClient() as client:
            if name == "gms_list_spaces":
                result = spaces.list_spaces(client).model_dump()
            elif name == "gms_list_recent":
                result = _list_recent_impl(
                    client,
                    space=arguments.get("space"),
                    limit=int(arguments.get("limit", 20)),
                    since=arguments.get("since"),
                )
            elif name == "gms_search":
                result = search.run_search(
                    client,
                    arguments["query"],
                    limit=int(arguments.get("limit", 20)),
                    space=arguments.get("space"),
                ).model_dump()
            elif name == "gms_fetch_post":
                result = _fetch_post_impl(client, arguments["url_or_slug"])
            elif name == "gms_video_transcript":
                result = _video_transcript_impl(client, arguments["url"])
            else:
                result = {"code": "unknown_tool", "name": name}
    except GmsError as e:
        result = _error_payload(e)
    except Exception as e:
        result = {"code": "runtime_error", "message": f"{type(e).__name__}: {e}"}
    return [TextContent(type="text", text=json.dumps(result, indent=2, default=str))]


async def amain() -> int:
    async with stdio_server() as (read, write):
        await server.run(read, write, server.create_initialization_options())
    return 0


def main() -> int:
    return asyncio.run(amain())


if __name__ == "__main__":
    sys.exit(main())
```
